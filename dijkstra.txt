import heapq

graph = {
    0: [(1,4),(2,1)],
    1: [(3,1)],
    2: [(1,2),(3,5)],
    3: []
}

def dijkstra(start):
    dist = {node: float('inf') for node in graph}
    dist[start] = 0

    heap = [(0, start)]

    while heap:
        d, node = heapq.heappop(heap)

        for nei, w in graph[node]:
            if d + w < dist[nei]:
                dist[nei] = d + w
                heapq.heappush(heap, (dist[nei], nei))

    return dist

print(dijkstra(0))
