class DynamicArray:
    def __init__(self):
        self.data = [None] * 2
        self.size = 0
        self.capacity = 2

    def resize(self):
        self.capacity *= 2
        new = [None] * self.capacity
        for i in range(self.size):
            new[i] = self.data[i]
        self.data = new

    def append(self, val):
        if self.size == self.capacity:
            self.resize()
        self.data[self.size] = val
        self.size += 1

    def insert(self, index, val):
        if self.size == self.capacity:
            self.resize()
        for i in range(self.size, index, -1):
            self.data[i] = self.data[i-1]
        self.data[index] = val
        self.size += 1

    def delete(self, index):
        for i in range(index, self.size-1):
            self.data[i] = self.data[i+1]
        self.size -= 1

    def search(self, val):
        for i in range(self.size):
            if self.data[i] == val:
                return i
        return -1
