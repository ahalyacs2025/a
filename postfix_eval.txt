#postfix evaluation
class Stack:
    def __init__(self):
        self.stack = []

    def push(self, item):
        self.stack.append(item)

    def pop(self):
        if self.stack:
            return self.stack.pop()

    def is_empty(self):
        return len(self.stack) == 0


def evaluate_postfix(postfix):
    stack = Stack()

    for char in postfix:
        if char.isdigit():
            stack.push(int(char))
        else:
            op2 = stack.pop()
            op1 = stack.pop()

            if char == '+':
                stack.push(op1 + op2)
            elif char == '-':
                stack.push(op1 - op2)
            elif char == '*':
                stack.push(op1 * op2)
            elif char == '/':
                stack.push(op1 // op2)

    return stack.pop()
expressions = ["23+", "782*+4*", "231*+9-"]

for expr in expressions:
    print(expr, "=", evaluate_postfix(expr))
